147  const double& alpha, const SDArray<NA>& a, const SDArray<NB>& b, SDArray<NC>& c)
148 {

150  int stride = std::accumulate(b.shape().begin(), b.shape().end(), 1, std::multiplies<int>());
151  // contraction list for thread parallelism
152  std::vector<DgemvArglist<NA, NB, NC>> task_list;
153  task_list.reserve(a.nnz());
154  // block contraction
155  for(int i = 0; i < nrows; ++i) {
156  // calc. range in 'a' contributes to c(i)
157  int a_lbound = i * stride;
158  int a_ubound = a_lbound + stride - 1;
159  typename SDArray<NA>::const_iterator ialo = a.lower_bound(a_lbound);
160  typename SDArray<NA>::const_iterator iaup = a.upper_bound(a_ubound);
161  if(ialo == iaup) continue;
162  if(!c.allowed(i)) continue;
163  // calc. find block in 'a' & 'b' contributes to c(i) to make argment list
164  DgemvArglist<NA, NB, NC> gemv_args;
165  for(typename SDArray<NA>::const_iterator ia = ialo; ia != iaup; ++ia) {
166  typename SDArray<NB>::const_iterator jb = b.find(ia->first % stride);
167  if(jb != b.end())
168  gemv_args.add(ia->second, jb->second);
169  }
170  // there's no blocks in 'a', 'b' contribute to c(i)
171  if(gemv_args.size() == 0) continue;
172  // allocate block element for c(i)
173  typename SDArray<NC>::iterator ic = c.reserve(i);
174  if(ic == c.end())
175  BTAS_THROW(false, "btas::thread_SDgemv required block could not be allocated");
176  // add argments to task_list
177  gemv_args.reset(ic->second, TransA, alpha, 1.0);
