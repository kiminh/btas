96 void Dgemv(const BTAS_TRANSPOSE& TransA,
97  const double& alpha, const DArray<NA>& a, const DArray<NB>& b, const double& beta, DArray<NC>& c) {
98  if(a.size() == 0 || b.size() == 0)
99  BTAS_THROW(false, "btas::Dgemv: array data not found");
100 
101  IVector<NC> c_shape;
102  gemv_contract_shape(TransA, a.shape(), b.shape(), c_shape);
103  // check and resize c
104  if(c.size() > 0) {
105  if(c_shape != c.shape())
106  BTAS_THROW(false, "btas::Dgemv: array shape of c mismatched");
107  }
108  else {
109  c.resize(c_shape);
110  c = 0.0;
111  }
112  // calling cblas_dgemv
113  int arows = std::accumulate(c_shape.begin(), c_shape.end(), 1, std::multiplies<int>());

115  if(TransA != NoTrans) std::swap(arows, acols);
116  cblas_dgemv(RowMajor, TransA, arows, acols, alpha, a.data(), acols, b.data(), 1, beta, c.data(), 1);
117 }

152 template<size_t NA, size_t NB, size_t NC>
153 void Dgemm(const BTAS_TRANSPOSE& TransA, const BTAS_TRANSPOSE& TransB,
154  const double& alpha, const DArray<NA>& a, const DArray<NB>& b, const double& beta, DArray<NC>& c) {
155  const size_t K = (NA + NB - NC)/2;
156  if(a.size() == 0 || b.size() == 0)
157  BTAS_THROW(false, "btas::Dgemm: array data not found");
158 
159  IVector<K> contracts;
160  IVector<NC> c_shape;
161  gemm_contract_shape(TransA, TransB, a.shape(), b.shape(), contracts, c_shape);
162  // check and resize c
163  if(c.size() > 0) {
164  if(c_shape != c.shape())
165  BTAS_THROW(false, "btas::Dgemm: array shape of c mismatched");
166  }
167  else {
168  c.resize(c_shape);
169  c = 0.0;
170  }
171  // calling cblas_dgemm
172  int arows = std::accumulate(c_shape.begin(), c_shape.begin()+NA-K, 1, std::multiplies<int>());
173  int acols = std::accumulate(contracts.begin(), contracts.end(), 1, std::multiplies<int>());
174  int bcols = std::accumulate(c_shape.begin()+NA-K, c_shape.end(), 1, std::multiplies<int>());
175  int lda = acols; if(TransA != NoTrans) lda = arows;
176  int ldb = bcols; if(TransB != NoTrans) ldb = acols;
177  cblas_dgemm(RowMajor, TransA, TransB, arows, bcols, acols, alpha, a.data(), lda, b.data(), ldb, beta, c.data(), bcols);
178 }

184  const IVector<NA>& a_shape = a.shape();

187  if(!std::equal(b_shape.begin(), b_shape.end(), a_shape.begin()+NA-NB))
188  BTAS_THROW(false, "Ddimd: array shape mismatched");

190  int nrows = std::accumulate(a_shape.begin(), a_shape.begin()+NA-NB, 1, std::multiplies<int>());
191  int ncols = b.size();
192  double* pa = a.data();
193  for(int i = 0; i < nrows; ++i) {
194  const double* pb = b.data();
195  for(int j = 0; j < ncols; ++j, ++pa, ++pb)
196  (*pa) *= (*pb);
