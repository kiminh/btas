37 void thread_QSDgesvd
38 (const BTAS_ARROW_DIRECTION& ArrowDir,
39  const QSDArray<2, Q>& a, SDArray<1>& s, QSDArray<2, Q>& u, QSDArray<2, Q>& vt, bool calc_full_u = false, bool calc_full_vt = false)
40 {
41  int nrows = a.shape(0);
42  int ncols = a.shape(1);
43  // Contraction list for thread parallelism
44  std::vector<DgesvdArglist<2, 2>> task_list;
45  task_list.reserve(a.size());
46  // Set task list
47  for(typename QSDArray<2, Q>::const_iterator ia = a.begin(); ia != a.end(); ++ia) {
48  if(ia->second->size() == 0) continue;
49  // Calc. block indices of U, S, V^T
50  int irow = ia->first / ncols;
51  int icol = ia->first % ncols;
52  int stag, utag, vtag;
53  if(ArrowDir == LeftArrow) {
54  stag = irow;
55  utag = irow * nrows + irow;
56  vtag = irow * ncols + icol;
57  }
58  else {
59  stag = icol;
60  utag = irow * ncols + icol;
61  vtag = icol * ncols + icol;
62  }
63  SDArray<1>::iterator is = s.reserve(stag);
64  SDArray<2>::iterator iu = u.reserve(utag);
65  SDArray<2>::iterator iv = vt.reserve(vtag);
66  DgesvdArglist<2, 2> gesvd_list(ia->second, is->second, iu->second, iv->second, calc_full_u, calc_full_vt);
67  task_list.push_back(gesvd_list);
68  }
69  parallel_call(task_list);
70 }
