template<typename T, size_t N, class Q = Quantum>
TVector<Qshapes<Q>, N> btas::QSTArray< T, N, Q >::m_q_shape
private

Quantum number indices.

m_q_shape[0] = { q[0][0], q[0][1], ... } (for the first rank) m_q_shape[1] = { q[1][0], q[1][1], ... } (for the second rank) ...
