98 template<size_t NA, size_t NB, size_t NC, class Q = Quantum>
99 void QSDgemv
100 (const BTAS_TRANSPOSE& TransA,
101  const double& alpha, const QSDArray<NA, Q>& a, const QSDArray<NB, Q>& b,
102  const double& beta, QSDArray<NC, Q>& c)
103 {
104  // Checking contraction quantum numbers
105  Q q_total;
106  TVector<Qshapes<Q>, NC> q_shape;
107  gemv_contract_qshape(TransA, a.q(), a.qshape(), b.q(), b.qshape(), q_total, q_shape);
108  TVector<Dshapes, NC> d_shape;
109  gemv_contract_dshape(TransA, a.dshape(), b.dshape(), d_shape);
110  // Checking shapes of c
111  if(c.size() > 0) {
112  if(q_total != c.q())
113  BTAS_THROW(false, "btas::QSDgemv: total quantum number of c mismatched");
114  if(q_shape != c.qshape())
115  BTAS_THROW(false, "btas::QSDgemv: quantum number indices of c mismatched");
116  if(d_shape != c.dshape())
117  BTAS_THROW(false, "btas::QSDgemv: block shape of c mismatched");
118  SDscal(beta, c);
119  }
120  else {
121  c.resize(q_total, q_shape, d_shape, false);
122  }
123  // Calling SDgemv
124  if(TransA == NoTrans)
125  thread_SDgemv(TransA, alpha, a, b, c);
126  else
127  thread_SDgemv(TransA, alpha, a.transposed_view(NB), b, c);
128 }
