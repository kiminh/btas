25 template<size_t N, size_t K>
26 void decompose_shape
27 (const IVector<K>& index, IVector<N>& a_permute)
28 {
29  std::set<int> iset(index.begin(), index.end());
30  if(iset.size() < K)
31  BTAS_THROW(false, "btas::decompose_shape: duplicate indices found in index");
32  if((*iset.begin()) < 0 || (*iset.rbegin()) >= N)
33  BTAS_THROW(false, "btas::decompose_shape: index is out-of-range");
34  int n = 0;
35  for(int i = 0; i < N; ++i)
36  if(iset.find(i) == iset.end())
37  a_permute[n++] = i;
38  for(int i = 0; i < K; ++i)
39  a_permute[i+N-K] = index[i];
40 }

43 void indexed_decompose_shape
44 (const IVector<NA>& a_symbols, const IVector<NL>& l_symbols, const IVector<NA-NL+2>& r_symbols,

55  for(int i = 0; i < NL; ++i) {
56  if(aset.find(l_symbols[i]) != aset.end())
57  lmap.insert(std::make_pair(l_symbols[i], nl++));
58  else
59  lshr = std::make_pair(l_symbols[i], i);
60  }
61  if(nl != lmap.size())
62  BTAS_THROW(false, "btas::indexed_decompose_shape: duplicate indices found in l_symbols");
63  if(nl != NL-1)
64  BTAS_THROW(false, "btas::indexed_decompose_shape: there's missing index in l_symbols");
65

67  std::map <int, int> rmap;
68  int nr = 0;
69  for(int i = 0; i < NR; ++i) {
70  if(aset.find(r_symbols[i]) != aset.end()) {
71  if(lmap.find(r_symbols[i]) != lmap.end())
72  BTAS_THROW(false, "btas::indexed_decompose_shape: found (new) factorized index in a_symbols");
73  rmap.insert(std::make_pair(r_symbols[i], nr++));
74  }
75  else {
76  rshr = std::make_pair(r_symbols[i], i);
77  }
78  }
79  if(nr != rmap.size())
80  BTAS_THROW(false, "btas::indexed_decompose_shape: duplicate indices found in r_symbols");
81  if(nr != NR-1)
82  BTAS_THROW(false, "btas::indexed_decompose_shape: there's missing index in r_symbols");
83 
84  if(lshr.first != rshr.first)
85  BTAS_THROW(false, "btas::indexed_decompose_shape: there's no shared index between l_symbols and r_symbols");
86 
87  int n = 0;
88  for(int i = 0; i < NA; ++i) {
89  typename std::map<int, int>::iterator ilm = lmap.find(a_symbols[i]);
90  if(ilm != lmap.end()) a_permute[ilm->second] = i;
91  typename std::map<int, int>::iterator irm = rmap.find(a_symbols[i]);
92  if(irm != rmap.end()) a_permute[irm->second+NL-1] = i;
93  }
94 
95  nl = lshr.second;
96  l_permute[nl] = NL-1;
97  for(int i = 0; i < nl; ++i) l_permute[i] = i;
98  for(int i = nl + 1; i < NL-1; ++i) l_permute[i] = i-1;
99

101  r_permute[nr] = 0;
102  for(int i = 0; i < nr; ++i) r_permute[i] = i+1;
103  for(int i = nr + 1; i < NR-1; ++i) r_permute[i] = i;
104 }

134 template<size_t NA, size_t K>
135 double decompose
136 (const DArray<NA>& a, const IVector<K>& index, DArray<NA-K+1>& l, DArray<K+1>& r, int D = 0)
137 {
138  const size_t NL = NA - K + 1;
139  const size_t NR = K + 1;
140 
141  IVector<NA> a_permute;
142  decompose_shape(index, a_permute);
143 
144  DArray<NA> a_copy;
145  Dpermute(a, a_permute, a_copy);
146 
147  DArray<1> s;
148  DArray<NL> l_tmp;
149  DArray<NL> r_tmp;
150  Dgesvd(a_copy, s, l_tmp, r_tmp);
151  Ddidm(s, r_tmp);
152  _lsv_truncate(l_tmp, l, D);
153  _rsv_truncate(r_tmp, r, D);
154  // Selected norm
155  double snorm = 0.0;
156  for(int i = 0; i < D; ++i) snorm += s(i) * s(i);
157  // Discarded norm
158  double dnorm = 0.0;
159  for(int i = D; i < s.size(); ++i) dnorm += s(i) * s(i);
160  // Return weight
161  return (snorm / (snorm + dnorm));
162 }
