112  template<size_t M>
113  void copy(const TSubArray<T, M>& a) {
114  // Calc. sub-array shape
115  IVector<M> a_shape;
116  for(int i = 0; i < M; ++i) a_shape[i] = a.m_upper_bound[i]-a.m_lower_bound[i]+1;
117  int a_size = std::accumulate(a_shape.begin(), a_shape.end(), 1, std::multiplies<int>());
118  assert(m_store->size() == a_size);
119  // If 0-dim. array
120  if(a_size == 0) return;
121  // Striding
122  const IVector<M>& a_stride = a.stride();
123  int ldt = a_shape[M-1];
124  // Get bare pointers
125  T* t_ptr = m_store->data();
126  const T* a_ptr = a.data();
127  // Copying elements
128  IVector<M> index(a.m_lower_bound);
129  int nrows = a_size / ldt;
130  for(int j = 0; j < nrows; ++j, t_ptr += ldt) {
131  int offset = dot(a_stride, index);
132  _fast_copy(ldt, a_ptr+offset, t_ptr);
133  for(int i = static_cast<int>(M)-2; i >= 0; --i) {
134  if(++index[i] <= a.m_upper_bound[i]) break;
135  index[i] = a.m_lower_bound[i];
136  }
137  }
138  }
